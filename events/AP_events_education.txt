namespace = AP_ESSM_A
country_event = {
	id = AP_ESSM_A.1
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		has_policy_flag = AP_education_stance_token
		has_policy_flag = AP_education_privledge_all
	}
	immediate = {
		remove_modifier = edu_all_token
		remove_modifier = edu_all_basic
		remove_modifier = edu_all_encouraged
		remove_modifier = edu_all_vital
		remove_modifier = edu_cit_token
		remove_modifier = edu_cit_basic
		remove_modifier = edu_cit_encouraged
		remove_modifier = edu_cit_vital
		remove_modifier = edu_elite_token
		remove_modifier = edu_elite_basic
		remove_modifier = edu_elite_encouraged
		remove_modifier = edu_elite_vital
		add_modifier = {
			modifier = edu_all_token
        }
	}
}

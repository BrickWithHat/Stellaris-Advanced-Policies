worker_educator = {
	category = worker
	condition_string = WORKER_JOB_TRIGGER
	clothes_texture_index = 3
	building_icon = building_disinformation_center

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	possible_precalc = can_fill_worker_job
	weight = {
		weight = @worker_job_weight
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 2
			has_living_standard = { type = living_standard_academic_privilege }
		}
		modifier = {
			factor = 3
			has_trait = trait_erudite
		}
		modifier = {
			factor = 2
			has_trait = trait_brainslug
		}
		modifier = {
			factor = 2
			OR = {
				has_trait = trait_intelligent
				has_trait = trait_natural_engineers
				has_trait = trait_natural_physicists
				has_trait = trait_natural_sociologists
			}
		}
		modifier = {
			factor = 0.1
			can_take_servant_job = yes
		}
		modifier = {
			factor = 0.65
			has_trait = trait_presapient_proles
		}
		modifier = {
			factor = 1.5
			has_job = researcher
		}
		modifier = {
			factor = 1.5
			has_trait = trait_latent_psionic
		}
		modifier = {
			factor = 2
			has_trait = trait_psionic
		}
	}
	resources = {
		category = planet_job_educator
		produces = {
			physics_research = 5
			society_research = 5
			engineering_research = 8
		}
		upkeep = {
			consumer_goods = 1
		}
	}
	planet_modifier = {
		planet_amenities_add = 3
		planet_jobs_physics_research_produces_mult = 0.01
		planet_jobs_society_research_produces_mult = 0.01
		planet_jobs_engineering_research_produces_mult = 0.025
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				OR = {
					has_valid_civic = civic_AP_good_education
					has_valid_civic = civic_AP_good_education_corporate
				}
			}
		}
		planet_amenities_add = 1
	}
}

specialist_tarrifs_clerk = {
	category = specialist
	condition_string = SPECIALIST_JOB_TRIGGER
	building_icon = building_commercial_zone
	clothes_texture_index = 2

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
		is_enslaved = no
	}
	possible_precalc = can_fill_specialist_job
	weight = {
		weight = @specialist_job_weight
		modifier = {
			factor = 2
			owner = {
				can_generate_trade_value = yes
			}
		}
		modifier = {
			factor = 1.5
			OR = {
				has_trait = trait_thrifty
				has_trait = trait_nuumismatic_administration
			}
		}
		modifier = {
			factor = 10
			owner = {
				is_megacorp = yes
			}
		}
		modifier = {
			factor = 2
			owner = {
				is_xenophile = yes
			}
		}
		modifier = {
			factor = 1.5
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		modifier = {
			factor = 0.5
			has_trait = trait_repugnant
		}
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 2
			solar_system = {
				has_system_trade_value >= 50
			}
		}
		modifier = {
			factor = 2
			solar_system = {
				has_system_trade_value >= 100
			}
		}
	}
	resources = {
		category = planet_job_tarrif_clerk
		produces = {
		}
		upkeep = {
			consumer_goods = 1.5
			energy = 2
		}
	}
	planet_modifier = {
		planet_amenities_add = 5
		trade_value_mult = 0.05
	}
	# Positive Amenities
	triggered_planet_modifier = {
		potential = {
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		modifier = {
			planet_amenities_add = 1
		}
	}
	# Negative Amenities
	triggered_planet_modifier = {
		potential = {
			OR = {
				has_trait = trait_repugnant
				has_trait = trait_robot_uncanny
			}
		}
		modifier = {
			planet_amenities_add = -1
		}
	}
}

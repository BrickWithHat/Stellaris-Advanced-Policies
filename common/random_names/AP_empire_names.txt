# Children (Democracies/Oligrachies)
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 100
			has_origin = AP_children_of_the_darkness
			OR = {
				has_authority = auth_democratic
				has_authority = auth_oligarchic
			}
			NOR = {
				has_civic = civic_AP_ancient_remnants
			}
		}
	}
	lookups = "<democratic_gen_desc> <AP_darkness> [This.Capital.GetName]"
	format = format.AP.children.1	# <democratic_gen_desc> [This.Capital.GetName] <AP_darkness>
	noun = format.AP.children.species.1	# [This.Capital.GetName] <AP_darkness>
}

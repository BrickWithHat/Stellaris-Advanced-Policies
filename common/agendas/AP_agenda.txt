#
# weight_modifier - scope = leader, from = country
# modifier
agenda_AP_admin_focus = {
	weight_modifier = {
		weight = 100
		modifier = {
			factor = 2
			from = {
				has_civic = civic_AP_advanced_administration
			}
		}
		modifier = {
			factor = 2
			leader_class = "governor"
		}
		modifier = {
			factor = 0.5
			leader_class = envoy
		}
	}
	modifier = {
		country_admin_cap_mult = 0.1
		leader_upkeep_empire_size_mult = -0.1
	}
}
